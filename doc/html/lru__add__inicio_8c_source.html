18/**
19 * @brief Añade un nodo dado al inicio de la lista doblemente enlazada (MRU).
20 *
21 * Esta función maneja cuatro casos principales:
22 * 1. Conexión del nuevo nodo con el antiguo head.
23 * 2. Actualización del puntero 'prev' del antiguo head.
24 * 3. Actualización del puntero 'head' del caché.
25 * 4. Manejo del caso en que la lista estaba vacía (actualización del tail).
26 *
27 * @param cache Puntero a la estructura LRUCache.
28 * @param node Puntero al nodo que se va a insertar como nuevo head.
29 * @return void
30 */
31
32void add_inicio(LRUCache* cache, Node* node)
33{
34 //Conecta el nuevo nodo con la estructura existente.
35 // Se conectan los nodos a ambas direcciones
36 node->next = cache->head;
37 node->prev = NULL; //El nuevo nodo siempre será el primero, por lo que su 'prev' es NULL.
38
39 //Si hay elementos, el antiguo head apunta al nuevo nodo
40 if (cache->head != NULL)
41 {
42 cache->head->prev = node;
43 }
44
45 // Actualizamos el head del cache
46 cache->head = node;
47
48 // si es el primer elemento, este tambien es el tail
49
50 if (cache->tail == NULL)
51 {
52 cache->tail = node;
53 }
54
55 //incrementamos el tamaño actual
56 cache->current_size++;
57}
