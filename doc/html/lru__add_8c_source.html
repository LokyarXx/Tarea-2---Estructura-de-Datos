19/**
20 * @brief Añade un dato a la caché LRU, siguiendo la política de reemplazo.
21 *
22 * Los casos manejados son:
23 * 1. **Cache Hit:** Si el dato existe, se remueve de su posición y se inserta al inicio (MRU).
24 * 2. **Cache Miss (Lleno):** Si el dato no existe y la caché está llena, se elimina el nodo
25 * LRU (tail), y luego se añade el nuevo nodo al inicio (MRU).
26 * 3. **Cache Miss (Hay espacio):** Si el dato no existe y hay espacio, se añade directamente al inicio (MRU).
27 *
28 * @param cache Puntero a la estructura LRUCache.
29 * @param data El carácter (letra) a añadir o usar.
30 * @return void No retorna valor.
31 */
32void add_data(LRUCache* cache, char data)
33{ //Verificación inicial del cache
34 if (cache == NULL || cache->capacity == 0)
35 {
36 printf("Error: El cache no ha sido inicializado.\n");
37 return;
38 }
39 //Revisa si existe el nodo en el cache
40 Node* existente = buscar_nodo(cache, data);
41 if (existente != NULL) //Ya existe el elemento
42 { //Se quita de su posición y se mueve al inicio
43 remove_node(cache, existente);
44 add_inicio(cache, existente);
45 printf("\nElemento %c ya existe, movido a inicio.\n", data);
46 return;
47 }
48 //Si el cache esta lleno, requiere reemplazo
49 if (cache->current_size >= cache->capacity)
50 { //Se identifica el LRU, que es el tail
51 Node* lru = cache->tail;
52 char dato_eliminado = lru->data;
53 //Se desvincula el nodo LRU y se libera su memoria.
54 remove_node(cache, lru);
55 free(lru);
56 printf("\nElemento %c eliminado (cache lleno)\n", dato_eliminado);
57 }
58 //Crea un nuevo nodo (aplica si habia espacio o si se acaba de liberar espacio)
59 Node* nuevo_nodo = create_node(data);
60 if(nuevo_nodo == NULL)
61 {
62 printf("Error al crear el nodo\n");
63 return;
64 }
65 //El nuevo nodo se añade al inicio (MRU)
66 add_inicio(cache, nuevo_nodo);
67 printf("\nElemento %c anyadido al cache.\n", data);
68}
