19/**
20 * @brief Busca un dato en la caché LRU y devuelve su posición (1-basada).
21 * * @param cache Puntero a la estructura LRUCache.}
22 * @param data El carácter (letra) a buscar.
23 * @return int La posición del dato (1-basada) si se encuentra, o -1 si no está.
24 */
25
26int search_cache(LRUCache* cache, char data)
27{
28 int position;
29 if (cache == NULL) //Valida si la cache ha sido creada
30 {
31 return -1;
32 }
33
34 Node* current = cache->head;
35 position = 1; // La posición se cuenta desde 1 (el head es la posición 1).
36
37 while (current != NULL) // Recorre la lista enlazada desde el MRU (head) hasta el LRU (tail).
38 {
39 if (current->data == data)
40 {
41 return position; //Dato encontrado, devuelve la posición.
42 }
43 current = current->next;
44 position++;
45 }
46
47 return -1; // Si no encuentra el dato, devuelve -1.
48}
