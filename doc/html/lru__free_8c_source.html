19/**
20 * @brief Libera toda la memoria dinámica asociada a la caché LRU.
21 *
22 * La liberación se realiza en dos fases:
23 * 1. Recorre la lista enlazada (desde el head) y libera cada nodo individualmente.
24 * 2. Finalmente, libera la estructura LRUCache principal.
25 * Se incluye una verificación para asegurar que la función pueda ser llamada incluso
26 * si la caché nunca fue creada.
27 *
28 * @param cache Puntero a la estructura LRUCache a liberar.
29 * @return void No retorna valor.
30 */
31void free_cache(LRUCache* cache)
32{
33 //Verificación de seguridad, no intenta liberar si el cache es NULL
34 if (cache == NULL)
35 {
36 return;
37 }
38 Node* current = cache->head;
39 Node* next_node = NULL;
40
41 //Recorre y libera todos los nodos
42 while (current != NULL)
43 {
44 next_node = current->next; // Guardamos el puntero, antes de liberar el nodo actual.
45 free(current); // Liberamos memoria del nodo actual.
46 current = next_node; // Avanzamos al siguiente nodo que guardamos.
47 }
48 free(cache);
49}
