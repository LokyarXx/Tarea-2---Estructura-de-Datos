18/**
19 * @brief Crea e inicializa una caché LRU con la capacidad dada.
20 *
21 * @param capacity Capacidad máxima del caché. Debe ser >= 5.
22 * @return LRUCache* Puntero a la caché creada, o NULL si falla
23 * (capacidad inválida o error de memoria).
24 */
25
26LRUCache* create_cache(int capacity){
27 if (capacity < 5){
28 printf("El tamanyo minimo del cache es 5.\n");
29 return NULL;
30 }
31
32 LRUCache* cache = (LRUCache*) malloc(sizeof(LRUCache));
33 if (cache == NULL) {
34 printf("Error al asignar memoria para el cache.\n");
35 return NULL;
36 }
37
38 cache->capacity = capacity;
39 cache->current_size = 0;
40 cache->head = NULL;
41 cache->tail = NULL;
42
43 return cache;
44}
