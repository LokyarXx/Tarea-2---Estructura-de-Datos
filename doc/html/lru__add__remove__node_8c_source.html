20/**
21 * @brief Remueve un nodo específico de la lista doblemente enlazada del caché.
22 *
23 * La remoción funciona ajustando los punteros del nodo anterior y posterior
24 * al nodo dado. También maneja los casos especiales donde el nodo a remover
25 * es el 'head' (MRU) o el 'tail' (LRU) de la caché.
26 *
27 * @param cache Puntero a la estructura LRUCache.
28 * @param node Puntero al nodo que se desea desvincular de la lista.
29 * @return void No retorna valor.
30 */
31void remove_node (LRUCache* cache, Node* node)
32{
33
34 //Manejo del puntero 'PREV' (Ajuste del nodo anterior)
35
36 if (node -> prev != NULL) //Si el nodo a remover no es el head
37 {
38 node->prev->next = node->next;// el anterior pasa al siguiente
39 }
40 else //Si el nodo a remover es el head, el siguiente nodo se convierte en el nuevo head (MRU)
41 {
42 cache->head = node->next;
43 }
44 if(node -> next != NULL) // Si el nodo a remover NO es el tail (tiene un nodo siguiente)
45 {
46 node->next->prev = node->prev; // el siguiente apunta al anterior
47 }
48 else
49 {
50 cache->tail = node->prev; // si es tail, el anterior es el nuevo tail
51 }
52 cache->current_size--;
53}
